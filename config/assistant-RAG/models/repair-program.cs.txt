// FILE: RepairProgram.cs
// MODEL: RepairProgram (Entity Framework)

using System;
using System.ComponentModel.DataAnnotations;
using System.ComponentModel.DataAnnotations.Schema;

namespace RepairManagerApi.Models
{
    public class RepairProgram
    {
        [Key]
        public int Id { get; set; }
        
        [Required]
        [StringLength(100)]
        public string Name { get; set; }
        
        [Required]
        public int RepairWorkflowId { get; set; }
        
        [ForeignKey("RepairWorkflowId")]
        public RepairWorkflow RepairWorkflow { get; set; }
    }
}
