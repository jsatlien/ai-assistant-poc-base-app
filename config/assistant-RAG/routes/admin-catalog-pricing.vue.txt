<template>
  <div class="catalog-pricing-page">
    <h1>Catalog Pricing</h1>
    <div class="page-actions">
      <button class="btn-primary" @click="openPriceModal()">Add Price</button>
      <div class="search-container">
        <input type="text" v-model="searchQuery" placeholder="Search catalog items..." class="search-input" />
      </div>
    </div>

    <div v-if="filteredPricing.length > 0" class="table-container">
      <table class="data-table">
        <colgroup>
          <col style="width: 25%">
          <col style="width: 15%">
          <col style="width: 20%">
          <col style="width: 20%">
          <col style="width: 20%">
        </colgroup>
        <thead>
          <tr>
            <th>Item Name</th>
            <th>Type</th>
            <th>Price</th>
            <th>Last Updated</th>
            <th>Actions</th>
          </tr>
        </thead>
        <tbody>
          <tr v-for="price in filteredPricing" :key="price.id">
            <td>{{ price.catalogItemName }}</td>
            <td>{{ price.catalogItemType }}</td>
            <td>${{ price.price.toFixed(2) }}</td>
            <td>{{ formatDate(price.updatedAt || new Date()) }}</td>
            <td>
              <div class="actions-cell">
                <button class="btn-edit" @click="editPrice(price)">Edit</button>
                <button class="btn-delete" @click="confirmDeletePrice(price)">Delete</button>
              </div>
            </td>
          </tr>
        </tbody>
      </table>
    </div>

    <div v-else class="empty-state">
      <p>No pricing information found. Add your first price to get started.</p>
    </div>

    <!-- Price Modal -->
    <div v-if="showPriceModal" class="modal">
      <div class="modal-content">
        <div class="modal-header">
          <h2>{{ isEditing ? 'Edit Price' : 'Add Price' }}</h2>
          <button class="close-btn" @click="closePriceModal">&times;</button>
        </div>
        <div class="modal-body">
          <form @submit.prevent="savePrice">
            <div class="form-group">
              <label for="catalogItem">Catalog Item</label>
              <select 
                id="catalogItem" 
                v-model="selectedCatalogItem" 
                class="form-control"
                required
                :disabled="isEditing"
              >
                <option value="" disabled>Select a catalog item</option>
                <optgroup v-for="type in catalogItemTypes" :key="type" :label="type">
                  <option 
                    v-for="item in getCatalogItemsByType(type)" 
                    :key="item.id" 
                    :value="item"
                  >
                    {{ item.name }}
                  </option>
                </optgroup>
              </select>
            </div>
            
            <div class="form-group">
              <label for="price">Price ($)</label>
              <input 
                id="price" 
                type="number" 
                step="0.01" 
                min="0" 
                v-model.number="currentPrice.price" 
                class="form-control"
                required
              />
            </div>
            
            <div class="form-actions">
              <button type="button" class="btn-secondary" @click="closePriceModal">Cancel</button>
              <button type="submit" class="btn-primary">Save</button>
            </div>
          </form>
        </div>
      </div>
    </div>

    <!-- Confirmation Modal -->
    <div v-if="showConfirmationModal" class="modal">
      <div class="modal-content confirmation-modal">
        <div class="modal-header">
          <h2>Confirm Delete</h2>
          <button class="close-btn" @click="closeConfirmationModal">&times;</button>
        </div>
        <div class="modal-body">
          <p>Are you sure you want to delete the price for "{{ priceToDelete?.catalogItemName }}"?</p>
          <div class="form-actions">
            <button class="btn-secondary" @click="closeConfirmationModal">Cancel</button>
            <button class="btn-delete" @click="deletePrice">Delete</button>
          </div>
        </div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: 'CatalogPricingManager',
  data() {
    return {
      searchQuery: '',
      showPriceModal: false,
      showConfirmationModal: false,
      isEditing: false,
      currentPrice: {
        id: null,
        catalogItemId: null,
        catalogItemType: '',
        catalogItemName: '',
        price: 0,
        updatedAt: null
      },
      selectedCatalogItem: '',
      priceToDelete: null,
      catalogItemTypes: ['Service', 'Part', 'Device']
    }
  },
  computed: {
    catalogPricing() {
      return this.$store.getters.getCatalogPricing
    },
    filteredPricing() {
      if (!this.searchQuery) return this.catalogPricing
      
      const query = this.searchQuery.toLowerCase()
      return this.catalogPricing.filter(price => 
        price.catalogItemName.toLowerCase().includes(query) || 
        price.catalogItemType.toLowerCase().includes(query)
      )
    },
    catalogItems() {
      return this.$store.getters.getAllCatalogItems
    }
  },
  created() {
    this.$store.dispatch('fetchCatalogPricing')
  },
  methods: {
    formatDate(date) {
      return new Date(date).toLocaleDateString('en-US', {
        year: 'numeric',
        month: 'short',
        day: 'numeric'
      })
    },
    getCatalogItemsByType(type) {
      return this.catalogItems.filter(item => item.type === type)
    },
    openPriceModal(price = null) {
      this.isEditing = !!price
      
      if (this.isEditing) {
        this.currentPrice = JSON.parse(JSON.stringify(price)) // Deep copy to avoid reference issues
        this.selectedCatalogItem = {
          id: price.catalogItemId,
          type: price.catalogItemType,
          name: price.catalogItemName
        }
      } else {
        this.currentPrice = {
          id: null,
          catalogItemId: null,
          catalogItemType: '',
          catalogItemName: '',
          price: 0,
          updatedAt: new Date()
        }
        this.selectedCatalogItem = ''
      }
      
      this.showPriceModal = true
    },
    closePriceModal() {
      this.showPriceModal = false
      this.currentPrice = {
        id: null,
        catalogItemId: null,
        catalogItemType: '',
        catalogItemName: '',
        price: 0,
        updatedAt: null
      }
      this.selectedCatalogItem = ''
    },
    async savePrice() {
      if (!this.selectedCatalogItem || this.currentPrice.price <= 0) {
        alert('Please select a catalog item and enter a valid price')
        return
      }

      try {
        // Update the current price object with catalog item details
        if (!this.isEditing) {
          this.currentPrice.catalogItemId = this.selectedCatalogItem.id
          this.currentPrice.catalogItemType = this.selectedCatalogItem.type
          this.currentPrice.catalogItemName = this.selectedCatalogItem.name
        }
        
        this.currentPrice.updatedAt = new Date()
        
        if (this.isEditing) {
          await this.$store.dispatch('updateCatalogPrice', this.currentPrice)
        } else {
          await this.$store.dispatch('addCatalogPrice', this.currentPrice)
        }
        
        this.closePriceModal()
      } catch (error) {
        console.error('Error saving price:', error)
        alert('Failed to save price. Please try again.')
      }
    },
    editPrice(price) {
      this.openPriceModal(price)
    },
    confirmDeletePrice(price) {
      this.priceToDelete = price
      this.showConfirmationModal = true
    },
    closeConfirmationModal() {
      this.showConfirmationModal = false
      this.priceToDelete = null
    },
    async deletePrice() {
      if (!this.priceToDelete) return

      try {
        await this.$store.dispatch('deleteCatalogPrice', this.priceToDelete.id)
        this.closeConfirmationModal()
      } catch (error) {
        console.error('Error deleting price:', error)
        alert('Failed to delete price. Please try again.')
      }
    }
  }
}
</script>
