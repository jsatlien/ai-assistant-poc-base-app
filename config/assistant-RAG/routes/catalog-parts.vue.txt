<template>
  <div class="parts-page">
    <div class="page-header">
      <h1 class="page-title">Parts</h1>
      <button class="btn-primary" @click="showAddPartModal = true">Add Part</button>
    </div>

    <div class="search-filter">
      <div class="search-box">
        <input 
          type="text" 
          v-model="searchQuery" 
          placeholder="Search parts..." 
          class="search-input"
        />
      </div>
    </div>

    <div v-if="filteredParts.length > 0" class="table-container">
      <table class="data-table">
        <thead>
          <tr>
            <th>SKU</th>
            <th>Name</th>
            <th>Manufacturer</th>
            <th>Device</th>
            <th>Description</th>
            <th>Actions</th>
          </tr>
        </thead>
        <tbody>
          <tr v-for="part in filteredParts" :key="part.id">
            <td>{{ part.sku }}</td>
            <td>{{ part.name }}</td>
            <td>{{ getManufacturerName(part.manufacturerId) }}</td>
            <td>{{ getDeviceName(part.deviceId) }}</td>
            <td>{{ part.description }}</td>
            <td class="actions-cell">
              <button class="btn-edit" @click="editPart(part)">Edit</button>
              <button class="btn-delete" @click="confirmDeletePart(part)">Delete</button>
            </td>
          </tr>
        </tbody>
      </table>
    </div>

    <div v-else class="empty-state">
      <p v-if="searchQuery">No parts found matching your search criteria.</p>
      <p v-else>No parts found. Add your first part to get started.</p>
      <button class="btn-primary" @click="showAddPartModal = true">Add Part</button>
    </div>

    <!-- Add/Edit Part Modal -->
    <div v-if="showAddPartModal || showEditPartModal" class="modal">
      <div class="modal-content">
        <div class="modal-header">
          <h2 class="modal-title">{{ showEditPartModal ? 'Edit Part' : 'Add New Part' }}</h2>
          <button class="modal-close" @click="closeModal">&times;</button>
        </div>
        <div class="modal-body">
          <div class="form-group">
            <label for="partSku">SKU</label>
            <input 
              type="text" 
              id="partSku" 
              v-model="currentPart.sku" 
              placeholder="Enter part SKU"
              class="form-control"
            />
          </div>
          <div class="form-group">
            <label for="partName">Part Name</label>
            <input 
              type="text" 
              id="partName" 
              v-model="currentPart.name" 
              placeholder="Enter part name"
              class="form-control"
            />
          </div>
          <div class="form-group">
            <label for="partManufacturer">Manufacturer <span class="required">*</span></label>
            <select 
              id="partManufacturer" 
              v-model="currentPart.manufacturerId" 
              class="form-control"
              required
            >
              <option value="" disabled>Select a manufacturer</option>
              <option v-for="manufacturer in manufacturers" :key="manufacturer.id || manufacturer.Id" :value="manufacturer.id || manufacturer.Id">
                {{ manufacturer.name || manufacturer.Name }}
              </option>
            </select>
          </div>
          <div class="form-group">
            <label for="partDevice">Device (Optional)</label>
            <select 
              id="partDevice" 
              v-model="currentPart.deviceId" 
              class="form-control"
            >
              <option :value="null">Not device-specific</option>
              <option v-for="device in devices" :key="device.id" :value="device.id">
                {{ device.name }}
              </option>
            </select>
          </div>
          <div class="form-group">
            <label for="partDescription">Description</label>
            <textarea 
              id="partDescription" 
              v-model="currentPart.description" 
              placeholder="Enter part description"
              class="form-control"
              rows="4"
            ></textarea>
          </div>
        </div>
        <div class="modal-footer">
          <button class="btn-secondary" @click="closeModal">Cancel</button>
          <button class="btn-primary" @click="savePart">Save</button>
        </div>
      </div>
    </div>
  </div>
</template>

<script>
import { mapGetters } from 'vuex';

export default {
  name: 'PartsCatalog',
  data() {
    return {
      searchQuery: '',
      showAddPartModal: false,
      showEditPartModal: false,
      currentPart: {
        id: null,
        name: '',
        description: '',
        sku: '',
        deviceId: null,
        manufacturerId: ''
      }
    };
  },
  computed: {
    ...mapGetters({
      parts: 'getParts',
      devices: 'getDevices',
      manufacturers: 'getManufacturers'
    }),
    filteredParts() {
      if (!this.searchQuery) {
        return this.parts;
      }
      
      const query = this.searchQuery.toLowerCase();
      return this.parts.filter(part => 
        part.name.toLowerCase().includes(query) || 
        part.description.toLowerCase().includes(query) ||
        (part.sku && part.sku.toLowerCase().includes(query)) ||
        this.getDeviceName(part.deviceId).toLowerCase().includes(query)
      );
    }
  },
  methods: {
    getManufacturerName(manufacturerId) {
      if (!manufacturerId) return 'N/A';
      const manufacturer = this.manufacturers.find(m => (m.id === manufacturerId || m.Id === manufacturerId));
      return manufacturer ? (manufacturer.name || manufacturer.Name) : 'N/A';
    },
    getDeviceName(deviceId) {
      if (!deviceId) return 'Not device-specific';
      const device = this.devices.find(d => d.id === deviceId);
      return device ? device.name : 'Unknown';
    },
    editPart(part) {
      this.currentPart = { ...part };
      this.showEditPartModal = true;
    },
    confirmDeletePart(part) {
      if (confirm(`Are you sure you want to delete ${part.name}?`)) {
        this.$store.dispatch('deletePart', part.id);
      }
    },
    closeModal() {
      this.showAddPartModal = false;
      this.showEditPartModal = false;
      this.currentPart = {
        id: null,
        name: '',
        description: '',
        sku: '',
        deviceId: null
      };
    },
    savePart() {
      if (!this.currentPart.name.trim()) {
        alert('Part name is required');
        return;
      }
      
      if (!this.currentPart.sku.trim()) {
        alert('SKU is required');
        return;
      }
      
      if (this.showEditPartModal) {
        // Update existing part
        this.$store.dispatch('updatePart', this.currentPart);
      } else {
        // Add new part
        const newPart = {
          ...this.currentPart,
          id: Date.now() // Temporary ID for demo
        };
        this.$store.dispatch('addPart', newPart);
      }
      
      this.closeModal();
    }
  },
  created() {
    // Fetch parts when component is created
    this.$store.dispatch('fetchParts');
    // Fetch devices for the dropdown
    this.$store.dispatch('fetchDevices');
    // Fetch manufacturers for the dropdown
    this.$store.dispatch('fetchManufacturers');
  }
};
</script>
